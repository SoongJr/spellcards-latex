
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% show the number of pages for each spell individually so user knows if they need to turn the card, or look for a second card with page 3 and 4
% code modified from \url{https://tex.stackexchange.com/a/578140/52439}
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\renewcommand*\pagemark{%
    \usekomafont{pagenumber}{\name\hfill\thepage~/~\letterlastpage}%
}
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% environment for formatting a spell card
% %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\newenvironment{card}{%
    % Note: the commands set by \DTLforeach are automatically accessible from this environment

    % Titlebar for spell (spell name and level)
    \begin{tabular}{p{\textwidth}} \toprule
        \section*{\huge\name\hfill\MakeUppercase{\class}~\level}\\
        \bottomrule
    \end{tabular} 
}{%
    \cleardoublepage% start new spell on a new card's _front_ after this spell is complete
    \setcounter{page}{1}% reset the page counter so every spell starts with 1
}
